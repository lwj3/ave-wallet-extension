检测钱包

  console.log('Ave Wallet detected!');
}

连接钱包

const accounts = await window.aveWallet.connect();
console.log('Connected:', accounts);

发送交易

const txHash = await window.aveWallet.sendTransaction({
  to: '0x...',
  value: '0x1000000000000000000', // 1 ETH
  gasPrice: '0x3b9aca00'
});
